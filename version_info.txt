# Métadonnées de version embarquées dans WebsiteBlocker.exe (PyInstaller --version-file).
# Un exécutable non signé qui expose des informations de version claires
# (éditeur, description, version, copyright) est nettement mieux accepté par
# Windows SmartScreen et la plupart des antivirus qu'un binaire « anonyme ».

  ffi=FixedFileInfo(
    filevers=(1, 1, 0, 0),
    prodvers=(1, 1, 0, 0),
    mask=0x3f,
    flags=0x0,
    OS=0x40004,
    fileType=0x1,
    subtype=0x0,
    date=(0, 0)
  ),
  kids=[
    StringFileInfo(
      [
        StringTable(
          u'040c04b0',
          [
            StringStruct(u'CompanyName', u'MultiHServices'),
            StringStruct(u'FileDescription', u'Gestionnaire de blocage de sites web'),
            StringStruct(u'FileVersion', u'1.1.0.0'),
            StringStruct(u'InternalName', u'WebsiteBlocker'),
            StringStruct(u'LegalCopyright', u'Copyright (c) 2026 MultiHServices'),
            StringStruct(u'OriginalFilename', u'WebsiteBlocker.exe'),
            StringStruct(u'ProductName', u'Website Blocker'),
            StringStruct(u'ProductVersion', u'1.1.0.0')
          ]
        )
      ]
    ),
    VarFileInfo([VarStruct(u'Translation', [1036, 1200])])
  ]
)
